Across the tree: the event bus#

Components with no DOM relationship talk through leanweb.eventBus — publish with dispatchEvent, subscribe with addEventListener, which returns the listener you later hand to removeEventListener. Bus callbacks run outside any DOM event, so they call update() themselves — as does the clock's setInterval.


            customElements.define('demo-clock',
               class extends LWElement {
                  constructor() {
                     super(ast);
                     setInterval(() => {
                        this.time = new Date().toLocaleTimeString();
                        leanweb.eventBus.dispatchEvent('tick', this.time);
                        this.update();
                     }, 1000);
                  }
               }
            );
         

            <p>Hearing <b lw>time ?? '-'</b></p>
            <button lw-bind:disabled='listener' lw-on:click='listen()'>Listen</button>
            <button lw-bind:disabled='!listener' lw-on:click='mute()'>Mute</button>
         

            customElements.define('demo-display',
               class extends LWElement {
                  listen() {
                     this.listener = leanweb.eventBus.addEventListener('tick', event => {
                        this.time = event.data;
                        this.update();
                     });
                  }

                  mute() {
                     leanweb.eventBus.removeEventListener(this.listener);
                     this.listener = null;
                  }
               }
            );
