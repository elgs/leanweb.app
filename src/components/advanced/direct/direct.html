A direct line: passing this#

Since lw-input: passes references, passing this hands the child the whole parent — methods included. Note the parent renders itself in add(): the click was the child's event, so the parent takes care of its own view.


            <p>Score <b lw>score</b></p>
            <demo-child lw-input:parent='this'></demo-child>
         

            customElements.define('demo-parent',
               class extends LWElement {
                  score = 0;

                  add(points) {
                     this.score += points;
                     this.update();
                  }
               }
            );
         

            <button lw-on:click='parent.add(10)'>+10 from the child</button>
         

            customElements.define('demo-child',
               class extends LWElement {
                  parent;
               }
            );
