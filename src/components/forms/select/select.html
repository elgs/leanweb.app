Selects#

A dropdown binds to one value; add multiple and the same directive binds to an array. Generate the options with lw-for.


            <select lw-model='zone'>
               <option lw lw-for='z in zones'>z</option>
            </select>
            <p>Meeting time in <b lw>zone</b></p>

            <select lw-model='crew' multiple>
               <option lw lw-for='p in people'>p</option>
            </select>
            <p lw>'On call: ' + crew.join(' + ')</p>
         

            class extends LWElement {
               zones = ['UTC', 'America/Phoenix', 'Asia/Tokyo'];
               zone = 'America/Phoenix';

               people = ['Ada', 'Grace', 'Linus'];
               crew = ['Ada'];
            }

Meeting time in zone

'On call: ' + crew.join(' + ')
