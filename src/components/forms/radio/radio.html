Radio buttons#

A radio group binds to one value. Set the property from code and the matching button lights up.


            <label lw-for='m in modes'>
               <input type='radio' lw-bind:value='m' lw-model='mode'> <span lw>m</span>
            </label>
            <button lw-on:click="mode = 'Transit'">Pick Transit</button>
            <p>Going by <b lw>mode.toLowerCase()</b></p>
         

            class extends LWElement {
               modes = ['Walk', 'Bike', 'Transit'];
               mode = 'Bike';
            }
