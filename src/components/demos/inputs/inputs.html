lw-input:#

Hand data from a parent to a child component. The child receives a reference to the very same object — not a copy — so either side sees the other's changes. Parent renders propagate to children automatically, and the child's own events re-render the child.


            <p><b lw>item.name</b> order</p>
            <button lw-on:click='item.qty++'>Add from parent</button>
            <demo-child lw-input:item='item'></demo-child>
         

            customElements.define('demo-parent',
               class extends LWElement {
                  item = { name: 'Espresso', qty: 1 };
               }
            );
         

            <p>Child sees <b lw>item?.qty</b></p>
            <button lw-on:click='item.qty++'>Add from child</button>
         

            customElements.define('demo-child',
               class extends LWElement {
                  item;
               }
            );
