lw-on:#

Listen to any DOM event. The handler is a plain expression with $event and $node in scope, and the component re-renders after it runs. Bind several events at once with a comma: lw-on:focus,blur.


            <button lw-on:click='count++'>Clicked <b lw>count</b> times</button>
            <input lw-on:focus,blur='seen($event, $node)' placeholder='focus me, leave me'>
            <p lw>last</p>
         

            class extends LWElement {
               count = 0;
               last = 'nothing seen yet';

               seen(event, node) {
                  this.last = event.type + ' on <' + node.localName + '>';
               }
            }
