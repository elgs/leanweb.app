Parent to listen on child's custom events

parent.html 

   <div class='parent'>
      <span>Parent</span>
      <span lw>quantity</span>
      <demo-child lw-on:add='onAdd($event)' </demo-child>
   </div>
parent.js 

   class extends LWElement {
      quantity = 0;
      
      onAdd(event) {
         this.quantity += event.detail;
      }
   }
child.html 

   <div class='child'>
      <span>Child</span>
      <div class='buttons'>
         <button lw-on:click='add(1)'>Add One</button>
         <button lw-on:click='add(-1)'>Remove One</button>
      </div>
   </div>
child.js 

   class extends LWElement {
      add(by) {
         this.dispatchEvent(new CustomEvent('add', {
            detail: by
         }));
      }
   }
