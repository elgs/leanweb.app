Working with 2D array

HTML 

   <button lw-on:click="reset()">Reset</button>
   <div lw-for="item, $i in array2d">
      <span lw-for="item1, $j in array2d[$i]">
         <input type="number" lw-model="array2d[$i][$j]">
         <span lw>item1</span>
      </span>
   </div>
Javascript 

   class extends LWElement {
      array2d = [
         [1, 2, 3],
         [4, 5, 6],
         [7, 8, 9],
      ];

      reset() {
         this.array2d.forEach(a => a.fill(0));
         this.update();
      }
   }
