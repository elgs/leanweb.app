Working with 2D array

HTML 

   <button lw-on:click='reset()'>Reset</button>
   <div lw-for='row in array2d'>
      <span lw-for='cell, $i in row'>
         <!-- cell is primitive type which is passed by value, -->
         <!-- so lw-model uses row[$i] which is passed by reference -->
         <input type='number' lw-model='row[$i]'>
         <span lw>cell</span>
      </span>
   </div>
Javascript 

   class extends LWElement {
      array2d = [
         [1, 2, 3],
         [4, 5, 6],
         [7, 8, 9],
      ];

      reset() {
         this.array2d.forEach(a => a.fill(0));
      }
   }
