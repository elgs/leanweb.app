lw-model: and lw-on:

HTML 

   <input type='text' lw-model='name'>
   <button lw-on:click='resetName($event, $node)'> Reset Name </button>
   <!-- You could bind multiple events -->
   <!-- <button lw-on:click,mouseout='resetName($event, $node)'> Reset Name </button> -->
   <span lw>name</span>
Javascript 

   class extends LWElement {
      resetName(event, node) {
         this.name = 'Leanweb';
      }
   }
