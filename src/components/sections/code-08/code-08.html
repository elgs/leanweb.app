Working with Radio Buttons

HTML 

   <button lw-on:click='chooseTwo()'>Choose Two</button>
   <div lw-for='item, $index in items'>
      <input type='radio' lw-bind:value='item' lw-model='picked'>
      <span lw>item</span>
   </div>
   <span lw>picked</span>
Javascript 

   class extends LWElement {
      items = ['one', 'two', 'three'];
      chooseTwo() {
         this.picked = 'two';
      }
   }
