Working with Select

HTML 

   <button lw-on:click='selectTwo()'>Select Two</button>
   <select lw-model="selectedOption">
      <option lw lw-for="item, $index in items">item</option>
   </select>
   <span lw> selectedOption </span>
Javascript 

   class extends LWElement {
      items = ['one', 'two', 'three'];
      selectTwo() {
         this.selectedOption = 'two';
         this.update();
      }
      selectedOption;
   }
