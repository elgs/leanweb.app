lw-input:

Pass data from a parent component down to a child using lw-input:. The child receives a reference to the same object, so changes made by either side are reflected in both components.

parent.html 

   <div class='parent'>
      <span>Parent</span>
      <span lw>item?.quantity</span>
      <button lw-on:click='update()'>Update Quantity</button>
      <demo-child lw-input:item='item'></demo-child>
   </div>
parent.js 

   class extends LWElement {
      item = {
         name: 'Banana',
         quantity: 1,
      };
   }
child.html 

   <div class='child'>
      <span>Child</span>
      <span lw>item?.quantity</span>
      <div class='buttons'>
         <button lw-on:click='addOne()'>Add One</button>
         <button lw-on:click='removeOne()'>Remove One</button>
      </div>
   </div>
child.js 

   class extends LWElement {
      addOne() {
         this.item.quantity += 1;
      }

      removeOne() {
         this.item.quantity -= 1;
      }
   }
