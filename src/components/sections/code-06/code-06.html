lw-input:

The following example shows how parent and child component share data by passing data with lw-input:.

parent.html 

   <div class="parent">
      <span>Parent</span>
      <span lw>item?.quantity</span>
      <div class="buttons">
         <button lw-on:click="updateQuantity()">Update Quantity</button>
      </div>
      <leanweb-demo-child lw-input:item='item'></leanweb-demo-child>
   </div>
parent.js 

   class extends LWElement {
      item = {
         name: 'Banana',
         quantity: 1,
      };

      // use the side effect that DOM is updated after `lw-on:` event is fired
      updateQuantity() {
      }
   }
child.html 

   <div class='child'>
      <span>Child</span>
      <span lw>item?.quantity</span>
      <div class="buttons">
         <button lw-on:click="addOne()">Add One</button>
         <button lw-on:click="removeOne()">Remove One</button>
      </div>
   </div>
child.js 

   class extends LWElement {
      addOne() {
         this.item.quantity += 1;
      }

      removeOne() {
         this.item.quantity -= 1;
      }
   }
