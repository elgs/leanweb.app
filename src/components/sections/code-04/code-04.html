HTML 

   <div lw lw-for='item, $index in items' lw-class:active='isActive($index)'>item</div>

      color: red;
   }
Javascript 

   class extends LWElement {
      items = ['one', 'two', 'three'];
      isActive(index) {
         return index === 1;
      }
   }
