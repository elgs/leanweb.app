Child to hold a reference to the parent

parent.html 

   <div class="parent">
      <span>Parent</span>
      <span lw>quantity</span>
      <demo-child lw-input:parent="this" </demo-child>
   </div>
parent.js 

   class extends LWElement {
      quantity = 0;
      
      add(by) {
         this.quantity += by;
         this.update();
      }
   }
child.html 

   <div class='child'>
      <span>Child</span>
      <div class="buttons">
         <button lw-on:click="add(1)">Add One</button>
         <button lw-on:click="add(-1)">Remove One</button>
      </div>
   </div>
child.js 

   class extends LWElement {
      add(by) {
         this.parent.add(by);
      }
   }
